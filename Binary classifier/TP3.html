def norm2(x, m, cov): 
    a = np.dot(np.transpose((x-m)), np.linalg.inv(cov)) 
    a = np.dot(a, (x-m)) 
    p =1/(np.sqrt(2*np.pi*np.linalg.det(cov)))*np.exp(-0.5*a) 
    return p
